28 // Total Size: 18 bytes
29 // BYTE | BIT layout (format is 'channel # . bit # of channel')
30 // 1 | c0.7 c0.6 c0.5 c0.4 c0.3 c0.2 c0.1 c0.0
31 // 2 | c1.4 c1.3 c1.2 c1.1 c1.0 c0.10 c0.9 c0.8
32 // 3 | c2.1 c2.0 c1.10 c1.9 c1.8 c1.7 c1.6 c1.5
33 // 4 | c2.9 c2.8 c2.7 c2.6 c2.5 c2.4 c2.3 c2.2
34 // 5 | c3.6 c3.5 c3.4 c3.3 c3.2 c3.1 c3.0 c2.10
35 // 6 | s2.1 s2.0 s1.1 s1.0 c3.10 c3.9 c3.8 c3.7
36 // 7 | mX (0-7)
37 // 8 | mX (8-15)
38 // 9 | mY (0-7)
39 // 10 | mY (8-15)
40 // 11 | mZ (0-7)
41 // 12 | mZ (8-15)
42 // 13 | m1 (0-7)
43 // 14 | m2 (0-7)
44 // 15 | KEY (0-7)
45 // 16 | KEY (8-15)
46 // 17 | RESERVED (wheel [0-7])
47 // 18 | RESERVED (wheel [5-7])
48 
49 // Channel | Physical Action | Value Ranges [low - '0' - high]
50 // c0 | Right Joystick X-Axis | [364 - 1024 - 1684]
51 // c1 | Right Joystick Y-Axis | [364 - 1024 - 1684]
52 // c2 | Left Joystick X-Axis | [364 - 1024 - 1684]
53 // c3 | Left Joystick Y-Axis | [364 - 1024 - 1684]
54 // s1 | Left Switch | [1 - 3]
55 // s2 | Right Switch | [1 - 3]
56 // mX | Mouse X-Axis | [-32768 - 0 - 32767]
57 // mY | Mouse Y-Axis | [-32768 - 0 - 32767]
58 // mZ | Mouse Z-Axis | [-32768 - 0 - 32767]
59 // m1 | Mouse Left Button | [0 - 1]
60 // m2 | Mouse Right Button | [0 - 1]
61 // key | Specific Key Presses | Bitmap: (bit #) (0 - 1)
62 // W: 0 Q: 4
63 // S: 1 E: 5
64 // A: 2 Shift: 6
65 // D: 3 Ctrl: 7
66 // wheel | Wheel Axis | [364 - 1024 - 1684]
