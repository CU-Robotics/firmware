27 // BYTE | BIT layout (format is 'channel # . bit # of channel')
28 // 1 | c0.7 c0.6 c0.5 c0.4 c0.3 c0.2 c0.1 c0.0

32 // 5 | c3.6 c3.5 c3.4 c3.3 c3.2 c3.1 c3.0 c2.10
33 // 6 | s2.1 s2.0 s1.1 s1.0 c3.10 c3.9 c3.8 c3.7
