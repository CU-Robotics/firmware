97  void solveSystem(float coeff[3][4], float output[3]) {
98  // Matrix d using coeff as given in cramer's rule
99  float d[3][3] = {
100  {coeff[0][0], coeff[0][1], coeff[0][2]},
101  {coeff[1][0], coeff[1][1], coeff[1][2]},
102  {coeff[2][0], coeff[2][1], coeff[2][2]}
103  };
104  // Matrix d1 using coeff as given in cramer's rule
105  float d1[3][3] = {
106  {coeff[0][3], coeff[0][1], coeff[0][2]},
107  {coeff[1][3], coeff[1][1], coeff[1][2]},
108  {coeff[2][3], coeff[2][1], coeff[2][2]},
109  };
110  // Matrix d2 using coeff as given in cramer's rule
111  float d2[3][3] = {
112  {coeff[0][0], coeff[0][3], coeff[0][2]},
113  {coeff[1][0], coeff[1][3], coeff[1][2]},
114  {coeff[2][0], coeff[2][3], coeff[2][2]},
115  };
116  // Matrix d3 using coeff as given in cramer's rule
117  float d3[3][3] = {
118  {coeff[0][0], coeff[0][1], coeff[0][3]},
119  {coeff[1][0], coeff[1][1], coeff[1][3]},
120  {coeff[2][0], coeff[2][1], coeff[2][3]},
121  };
122 
123  // Calculating Determinant of Matrices d, d1, d2, d3
124  float D = determinantOfMatrix(d);
125  float D1 = determinantOfMatrix(d1);
126  float D2 = determinantOfMatrix(d2);
127  float D3 = determinantOfMatrix(d3);
128 
129  // Case 1
130  if (D != 0) {
131  // Coeff have a unique solution
132  output[0] = D1 / D;
133  output[1] = D2 / D;
134  output[2] = D3 / D;
135  }
136  // Case 2
137  else {
138  if (D1 == 0 && D2 == 0 && D3 == 0) {
139  output[0] = 0;
140  output[1] = 0;
141  output[2] = 0;
142  } else if (D1 != 0 || D2 != 0 || D3 != 0) {
143  output[0] = 0;
144  output[1] = 0;
145  output[2] = 0;
146  }
147  }
