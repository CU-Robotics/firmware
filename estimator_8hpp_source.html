96  void solveSystem(float coeff[3][4], float output[3]) {
97  // Matrix d using coeff as given in cramer's rule
98  float d[3][3] = {
99  {coeff[0][0], coeff[0][1], coeff[0][2]},
100  {coeff[1][0], coeff[1][1], coeff[1][2]},
101  {coeff[2][0], coeff[2][1], coeff[2][2]}
102  };
103  // Matrix d1 using coeff as given in cramer's rule
104  float d1[3][3] = {
105  {coeff[0][3], coeff[0][1], coeff[0][2]},
106  {coeff[1][3], coeff[1][1], coeff[1][2]},
107  {coeff[2][3], coeff[2][1], coeff[2][2]},
108  };
109  // Matrix d2 using coeff as given in cramer's rule
110  float d2[3][3] = {
111  {coeff[0][0], coeff[0][3], coeff[0][2]},
112  {coeff[1][0], coeff[1][3], coeff[1][2]},
113  {coeff[2][0], coeff[2][3], coeff[2][2]},
114  };
115  // Matrix d3 using coeff as given in cramer's rule
116  float d3[3][3] = {
117  {coeff[0][0], coeff[0][1], coeff[0][3]},
118  {coeff[1][0], coeff[1][1], coeff[1][3]},
119  {coeff[2][0], coeff[2][1], coeff[2][3]},
120  };
121 
122  // Calculating Determinant of Matrices d, d1, d2, d3
123  float D = determinantOfMatrix(d);
124  float D1 = determinantOfMatrix(d1);
125  float D2 = determinantOfMatrix(d2);
126  float D3 = determinantOfMatrix(d3);
127 
128  // Case 1
129  if (D != 0) {
130  // Coeff have a unique solution
131  output[0] = D1 / D;
132  output[1] = D2 / D;
133  output[2] = D3 / D;
134  }
135  // Case 2
136  else {
137  if (D1 == 0 && D2 == 0 && D3 == 0) {
138  output[0] = 0;
139  output[1] = 0;
140  output[2] = 0;
141  } else if (D1 != 0 || D2 != 0 || D3 != 0) {
142  output[0] = 0;
143  output[1] = 0;
144  output[2] = 0;
145  }
146  }
